      def initialize(type,base_name,version="_54")
        @type = type
        ToolDB::DBConnection.connect(version)
        @organisms = Hash.new
        ToolDB::Organism.find(:all).each do |organism|
          @organisms[organism.name] = [ 0, 0 ,0 ,0]
        end
        @base_name = base_name
        @version = version
      end
