      def get_clustalw
        
        answer = []
        
        cigar = self.vulgar_line.split(",")

        target_seq = self.target.protein_seq[self.target_start..self.target_stop]
        query_seq = self.query.protein_seq[self.query_start..self.query_stop]
        
        target_aln = ""
        query_aln = ""
        
        cigar.each do |c|
          c = c.split(" ")
          char = c.shift
          
          target_operation = c.shift.to_i
          query_operation = c.shift.to_i
          
          if char == "M"
            target_operation.times do
              target_aln += target_seq.slice!(/^[A-Za-z]/)
            end
            query_operation.times do
              query_aln += query_seq.slice!(/^[A-Za-z]/)
            end
          elsif char == "G"
            target_aln += ("-"*target_operation)
            query_aln += ("-"*query_operation)
          else
            raise "Unknown operation (#{char})"
          end
          
        end
        
        target_aln += target_seq
        query_aln += query_seq
        
        answer << Bio::FastaFormat.new(Bio::Sequence::AA.new(target_aln).to_fasta(self.target.stable_id))
        answer << Bio::FastaFormat.new(Bio::Sequence::AA.new(query_aln).to_fasta(self.query.stable_id))
        return Bio::Alignment::OriginalAlignment.new(answer)
        
      end
