      def phylip

        answer = Array.new
        tree = self.tree
        t = self.plain_tree
        taxa = Hash.new
        counter = 1
        
        tree.each_node do |node|

          unless node.inspect.match(/[0-9]/)
            taxa[node.to_s] = "PF000#{counter}"
            t.gsub!("#{node.to_s.gsub(/\s/, '_')}", "#{taxa.fetch(node.to_s)}")
            counter += 1
          end

        end
        
        @code = taxa
        return t

      end
