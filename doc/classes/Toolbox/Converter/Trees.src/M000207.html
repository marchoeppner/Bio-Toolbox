      def from_phylip(key)
        
        @keys = Hash.new
        
       IO.foreach(key) do |line|
          elements = line.strip.split(",")
          @keys[elements[1]] = elements[0].gsub(/[()]/, '')
        end
        
        answer = self.plain_tree

        @keys.each do |code,ordinary|
          answer.gsub!(code, ordinary)
        end
          
        return answer
        
      end
