      def convert
        
        new_header = []
        @rows.each_key { |k| new_header.push(k.strip) }
        new_rows = @header
        puts ",#{new_header.join(",")}"
        new_rows.each do |row|
          print "#{row.strip} - "
          @rows.each do |k,values|
            print "#{values.shift},"
            @rows[k] = values
          end
          print "\n"  
        end
      end
