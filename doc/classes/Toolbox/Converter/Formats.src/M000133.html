      def fasta_to_phylip
        
        ff = Bio::FastaFormat.open(infile)
        counter = 0
        
        ff.each do |entry|
          
          counter += 1
          if counter < 10
            zeros = "0000000"
          elsif counter >= 10
            zeros = "000000"
          elsif counter >= 100
            zeros = "00000"
          elsif counter >= 1000
            zeros = "0000"
          end
          name = "PF#{zeros}#{counter}"
          
          @code[entry.definition] = name
          
          puts ">#{name}"
          puts "#{entry.aaseq}"

        end
        
        k = File.new("#{@infile}_key.txt", "a")
        @code.each do |key,taxon|
          k.puts "#{key},#{taxon}"
        end
        k.close
        
      end
