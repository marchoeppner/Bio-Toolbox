      def biograph_coordinates(type="nt")
        
        answer = []
        type == "peptide" ? cig = "#{self.peptide_cigar_line}" : cig = "#{self.nt_cigar_line}"
        
        position = 1
        
        until cig.length == 0
          
          x = cig.slice!(/^\d+/).to_i
          char = cig.slice!(/^[A-Z]/)
          
          if char == "M"
            answer << "#{position}..#{position+x}"
            position += x
          else
            position += x
          end
          
        end
        
        return answer
        
                        end
