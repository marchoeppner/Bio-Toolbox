      def region_is_aligned?(hit)
        hit_coords = [ hit.target_start.to_i , hit.target_end.to_i ].sort
        hit_start = hit_coords.shift
        hit_stop = hit_coords.shift
        
        self.homology_alignments.each do |align|
          #puts align.inspect
          
          #puts "Checking for overlap #{align.query.stable_id} (#{hit.target_def})"
          if hit_start >= align.query_start and hit_stop <= align.query_stop
           # puts "#{align.query_start}/#{hit_start} - #{align.query_stop}/#{hit_stop}"
            return true
          end
        end
        
        return false
        
      end
