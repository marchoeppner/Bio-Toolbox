      def map_peptide_position(pos,verbose=false)
      
        cig = "#{self.peptide_cigar_line.clone}"
        
        #puts "\tMap: Peptide length: #{self.member.peptide_seq.length}"
        #puts "\tMap: Aligned peptide length: #{self.aligned_seq.length}"
        
        gapped_counter = 0
        counter = 0
        
        until cig.length == 0
                
                i = cig.slice!(/^\d+/).to_i
                char = cig.slice!(/^\w/)
                
                i.times do 
                
                        if char == "D"
                                gapped_counter += 1
                        else
                                gapped_counter += 1
                                counter += 1
                        end
                        
                        return gapped_counter if counter == pos
                        
                end
        
        end
        
        puts "Didn't meet the end... #{gapped_counter},#{counter} where position was #{pos}" if verbose
        return gapped_counter
      
      end
