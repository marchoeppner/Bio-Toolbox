      def exon_overlaps?(hit)
        
        if self.member.strand == true
          start = hit.target_start+self.member.start
          stop = hit.target_end+self.member.start
        else
          start = self.member.stop-hit.target_end
          stop = self.member.stop-hit.target_start
        end
        
        overlap = true
        
        self.member.transcripts.each do |transcript|
          
          this_overlap = false
          
          transcript.exons.each do |exon|
            if start >= exon.start and stop <= exon.stop
              this_overlap = true
            end
          end
          
          overlap = false if this_overlap == false
          
        end
        
        return overlap
        
      end
