      def aligned_position(feature,verbose=false)
                
        answer = []
        
        puts "\tMapping feature onto: #{self.member.start}/#{self.member.stop}/#{self.member.strand}"  if verbose
              
        puts "\t>Feature: #{feature.stable_id} (#{feature.start}/#{feature.stop}/#{feature.strand})" if verbose
          
        this_answer = []
          
        cig = "#{self.nt_cigar_line}"
        
        raise "No nt cigar line found for #{self.member.stable_id}" if cig.length == 0
          
        start = feature.start-self.member.start
        stop = feature.stop-self.member.start
          
        gapped_pos = 0
        ungapped_pos = 0
          
        until ungapped_pos == stop
            
                this_char = cig.slice!(0..0)
            
          if this_char == "-"
            gapped_pos += 1
          else
            gapped_pos += 1
            ungapped_pos += 1
          end
            
          this_answer << gapped_pos if ungapped_pos == start
            
        end
          
        this_answer << gapped_pos
        puts "\t>>Mapped to #{answer.join("..")}" if verbose
        answer << this_answer.join("..")
          
        return answer
        
      end
