                        def infernal_hits(threshold=3.0)
                        
                                hits = self.member.infernal_hits.select{|hit| hit.score.to_f >= threshold}
                                
                                infernal_groups = []
                                
                                hits.each do |hit|
                                        
                                        add = true
                                        
                                        infernal_groups.each do |group|
                                        
                                                next if hit.strand != group.strand or add == false
                                                
                                                if hit.strand == 1
                                                        add = false if hit.member_start < group.stop and hit.member_stop > group.start
                                                else
                                                        add = false if hit.member_start > group.stop and hit.member_stop < group.start           
                                                end

                                                group.members << hit unless add
                                                
                                        end
                                        
                                        infernal_groups << InfernalGroup.new(hit) if add
                                
                                end
                                
                                answer = []
                                
                                infernal_groups.each do |group|
                                
                                        answer << group.members.sort_by{|g| g.score }[0]
                                        
                                end
                                
                                return answer.sort_by{|h| h.member_start}

                        end
