                        def get_aligned_track(my_panel)
                                
                                track = my_panel.add_track("#{self.stable_id} (#{self.organism_name}]", :label => false, :glyph => { 'gap' => :box, 'aln' => :generic}, :colour =>  [0,0,1])
                                
                                coordinates = self.biograph_coordinates(type="peptide")
                                coordinates.unshift("0..1")
                                features = []
                                gap_starts = []
                                gap_stops = []
                                
                                first = coordinates.shift
                                features << Bio::Feature.new('aln', "#{first}")
                                
                                intron_start = first.split("..")[1]
                                
                                coordinates.each do |coord|
                                        intron_stop = coord.split("..")[0]
                                        features << Bio::Feature.new('gap',"#{intron_start}..#{intron_stop}")
                                        features << Bio::Feature.new('aln', "#{coord}")
                                end
                                
                                composite = Bio::Feature.new('aln', "join(#{coordinates.join(',')})", [], nil, features)
                                
                                track.add_feature(composite, :label => "composite")
                                
                                return my_panel
                                
                        end
