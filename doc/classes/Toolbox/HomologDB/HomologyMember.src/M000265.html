      def fetch_overlapping_alignments(start,stop)
        
        answer = []
        
        self.homology_alignments.select{|a| a.analysis == "local"}.each do |align|
          
          if align.target_start <= start and align.target_stop >= stop
            answer << align
          end
          
        end
        
        return answer
        
      end
