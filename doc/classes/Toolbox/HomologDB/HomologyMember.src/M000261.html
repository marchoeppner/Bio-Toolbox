      def ncrna_overlaps?(hit)
        
        if self.member.strand == true
          start = hit.target_start+self.member.start
          stop = hit.target_end+self.member.start
        else
          start = self.member.stop-hit.target_end
          stop = self.member.stop-hit.target_start
        end
        
        self.member.ncrna_members.each do |ncrna|
          
          if start >= ncrna.start and stop <= ncrna.stop or start <= ncrna.stop and stop >= ncrna.stop or start < ncrna.start and stop > ncrna.start
            return "#{ncrna.stable_id}"
          end
        
        end
        
        return false
        
      end
