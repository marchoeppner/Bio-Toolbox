      def fetch_ner_regions(start,stop)
        
        answer = []
        
        aligns = self.homology_alignments.select{|a| a.analysis == "ner" and a.target_start <= start and a.target_stop >= stop }
        aligns.each do |align|
          coords = align.get_ner_coords
          coords.each do |ner_start,ner_stop|
            answer << [ner_start,ner_stop] if ner_start <= start and ner_stop >= stop
          end
        end
        
        return answer
      
      end
