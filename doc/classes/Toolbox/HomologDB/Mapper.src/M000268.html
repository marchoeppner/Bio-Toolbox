      def _map
        
        alignment = self.alignment_slice.get_clustalw
        
        position = self.alignment_slice.homology_alignment.target_start
                
        alignment.each_site do |column|
          
          target_nt = column.shift
          query_nt = column.shift

          if target_nt == "-" 
            @aligned_target_start += 1 if position <= self.target_start
            @aligned_target_stop += 1 if position <= self.target_stop
          else
            position += 1
            @aligned_target_start += 1 if position <= self.target_start
            @aligned_target_stop += 1 if position <= self.target_stop
          end
          
          if query_nt == "-"
            @aligned_query_start += 1 if position <= self.target_start
            @aligned_query_stop += 1 if position <= self.target_stop
          else
            @aligned_query_start += 1 if position <= self.target_start
            @aligned_query_stop += 1 if position <= self.target_stop
            @query_start += 1 if position <= self.target_start
            @query_stop += 1 if position <= self.target_stop
          end
          
        end
        
      end
