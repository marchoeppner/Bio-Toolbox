      def _create_underlying_alignments
        
        aln = self.homology_alignment.get_clustalw
        
        aln.each_pair do |name,seq|
          
          if name == self.target.stable_id
            @aligned_target_seq = seq
          elsif name == self.query.stable_id
            @aligned_query_seq = seq
          else
            raise "Sequence name not reckognized (#{name})"
          end
          
        end
      
      end
