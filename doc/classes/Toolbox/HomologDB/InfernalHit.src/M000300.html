                def to_fasta
                        
                        if self.strand == -1
                                seq = self.member.nucleotide_sequence.sequence[self.member_start..self.member_stop]
                                return Bio::FastaFormat.new(Bio::Sequence::NA.new(seq).complement.to_fasta("#{self.member.stable_id}|infernal|#{self.score}|#{self.rfam.accession}"))
                        else
                                return Bio::FastaFormat.new(Bio::Sequence::NA.new(self.member.nucleotide_sequence.sequence[self.member_start..self.member_stop]).to_fasta("#{self.member.stable_id}|Infernal|#{self.score}|#{self.rfam.accession}"))
                        end
                        
                end
