      def do_local_align(start=0,stop=self.alignment_length)
        
        cigar = self.cigar_line.split(",")

        answer = []
        
        target = self.target.seq.downcase[self.target_start..self.target_stop]
        query = self.query.seq.downcase[self.query_start..self.query_stop]
        
        #puts "#{target[0..50]} | #{self.target_strand} / #{self.target.strand}"
        #puts "#{query[0..50]} | #{self.query_strand} / #{self.query.strand}"
        
        gapped_target = ""
        gapped_query = ""
        
        gapped_start = 0
        gapped_stop = 0
        curr_count = 0
        
        cigar.each do |c|
          
          operation = c.slice!(/^[A-Z]/)
          count = c.strip.slice(/[0-9]*/).to_i

          if operation == "M"
            
            count.times do
              curr_count += 1
              gapped_target += target.slice!(/^[a-z]/)
              gapped_query += query.slice!(/^[a-z]/)
            end
            
          elsif operation == "I"
            
            count.times do
              gapped_start += 1 unless curr_count >= start
              gapped_stop += 1 unless curr_count >= stop
              gapped_target += "-"
              gapped_query += query.slice!(/^[a-z]/)
            end
            
          elsif operation == "D"
            
            count.times do 
              curr_count += 1
              gapped_target += target.slice!(/^[a-z]/)
              gapped_query += "-"
            end
          
          else
            raise "Unknown operation: #{operation}!"
          end
          
          
        end

        if start == 0
          
          answer << Bio::FastaFormat.new(Bio::Sequence::NA.new(gapped_target).to_fasta("#{self.target.stable_id}"))
          answer << Bio::FastaFormat.new(Bio::Sequence::NA.new(gapped_query).to_fasta("#{self.query.stable_id}"))
          return Bio::Alignment::OriginalAlignment.new(answer)
        
        else
          
          gapped_start = gapped_start-self.target_start+start
          gapped_stop = gapped_stop-self.target_start+stop

          answer << Bio::FastaFormat.new(Bio::Sequence::NA.new(gapped_target[gapped_start..gapped_stop]).to_fasta("#{self.target.stable_id}"))
          answer << Bio::FastaFormat.new(Bio::Sequence::NA.new(gapped_query[gapped_start..gapped_stop]).to_fasta("#{self.query.stable_id}"))
          return Bio::Alignment::OriginalAlignment.new(answer)
          
        end
      end
