      def do_ner_align
        
        cigar = self.cigar_line.split(",")
        
        if "#{self.target_strand}" == "-1"
         # puts "complement of target"
          target = self.homology_member.homology_group.member.to_fasta.data[self.target_stop.to_i..self.target_start.to_i]
          target = Bio::Sequence::NA.new(target).complement.seq.strip
        else
          target = self.homology_member.homology_group.member.to_fasta.data[self.target_start.to_i..self.target_stop.to_i].strip
        end
        
        
        if "#{self.query_strand}" == "-1"
          query = self.homology_member.member.sequence.sequence[self.query_stop.to_i..self.query_start.to_i]
          query = Bio::Sequence::NA.new(query).complement.seq.strip
        else
          query = self.homology_member.member.sequence.sequence[self.query_start.to_i..self.query_stop.to_i].strip
        end
  
        p_target = ""
        p_query = ""
                
        cigar.each do |cig|
          
          elements = cig.split(" ")
          
          operation = elements.shift
          
          query_count = elements.shift.to_i
          target_count = elements.shift.to_i
          
          if operation == "M"
            
            query_count.times do
              p_query += query.slice!(/^[A-Za-z]/)
            end
            target_count.times do 
              p_target += target.slice!(/^[A-Za-z]/)
            end
            
          elsif operation == "I"
            
            query_count.times do 
              p_target += target.slice!(/^[A-Za-z]/)
            end
            target_count.times do
              p_query += "-"
            end
            
          elsif operation == "G"
            
            raise "Error - both gap counts > zero" if target_count != 0 and query_count != 0
            
            target_count.times do 
              p_target += target.slice!(/^[A-Za-z]/)
              p_query += "-"
            end
              
            query_count.times do 
              p_target += "-"
              p_query += query.slice!(/^[A-Za-z]/)
            end
            
          elsif operation == "N"
            
            target_count.times do
              target.slice!(/^[A-Za-z]/)
            end
            query_count.times do
              query.slice!(/^[A-Za-z]/)
            end
            p_target += "|gap|"
            p_query += "|gap|"
              
          else
              puts "unknown operation (#{operation})"
          end
       
        end
        answer = []
        answer << Bio::FastaFormat.new(Bio::Sequence::NA.new(p_target).to_fasta(self.target.stable_id))
        answer << Bio::FastaFormat.new(Bio::Sequence::NA.new(p_query).to_fasta(self.query.stable_id))
        return Bio::Alignment::OriginalAlignment.new(answer)
        
      end
