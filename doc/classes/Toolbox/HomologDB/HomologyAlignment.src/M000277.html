      def get_mapper(ncrna)
        
        return nil if self.analysis == "ner" or self.is_aligned?(ncrna) == false
        
        internal_start = ncrna.hostgene_start
        internal_stop = ncrna.hostgene_stop
        
        puts "ncRNA #{ncrna.start}/#{ncrna.stop}/#{ncrna.strand}"
        puts "Host: #{ncrna.member.start}/#{ncrna.member.stop}/#{ncrna.member.strand}"
        puts "Internal: #{ncrna.hostgene_start}/#{ncrna.hostgene_stop}"
        
        aln = self.get_clustalw
        
        aln.each_pair do |name,seq|
          
          next unless "#{name}" == "#{self.target.stable_id}"
          puts "Matching names, starting mapping process..."
          
          count = 0
          start_count = 0
          stop_count = 0
          
          until count == internal_stop or seq.length == 0
            
            char = seq.downcase.slice!(/^[a-z-]/)
            count += 1 unless "#{char}" == "-" or char =
            start_count += 1 unless count > internal_start or internal_start < self.target_start
            stop_count += 1
            
          end
          
          puts "Returning coordinates [#{start_count} , #{stop_count}]"
          return [ start_count , stop_count ]
          
        end 
            
        raise "IDs not matching... aborting mapping"
        
      end
