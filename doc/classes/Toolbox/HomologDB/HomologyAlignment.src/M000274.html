      def get_ner_coords
        
        cigar = self.cigar_line.split(",")
        
        ner_coords = []
        
        loc = 0 + self.target_start
        
        cigar.each do |c|
          
          elements = c.strip.split(" ")
          
          operation = elements.shift
          target_count = elements.shift.to_i
          query_count = elements.shift.to_i
          
          if operation == "N"
            
            ner_coords << [ loc, loc + target_count ]
            loc += target_count
            
          else
          
            loc += target_count
            
          end
          
        end
        
        return ner_coords
        
      end
