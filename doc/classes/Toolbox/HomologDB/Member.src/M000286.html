          def get_genomic_region(range)
                
                self.organism.connect_to_genome
                
                gene = Ensembl::Core::Gene.find_by_stable_id(self.stable_id)
                slice = Ensembl::Slice.new(gene.seq_region,gene.start-range,gene.stop+range,gene.strand)
                
                return slice.seq
          
          end
