      def self.connect(version="")

        establish_connection(
                              :adapter => Homolog_DB_ADAPTER,
                              :host => Homolog_DB_HOST,
                              :database => "#{Homolog_DATABASE}#{version}",
                              :username => Homolog_DB_USERNAME,
                              :password => Homolog_DB_PASSWORD
                              #:port => port
                            )
      end
