      def taxon_list
        
        answer = Array.new
        answer.push("\n")
        answer.push("BEGIN TAXA;")
        answer.push("\tTITLE #{@title};")
        answer.push("\tDIMENSIONS NTAX=#{@taxa.keys.nitems};")
        answer.push("\tTAXLABELS")
        answer.push("\t\t#{@taxa.keys.collect{|o| o.gsub(/\s/, '_')}.join(" ")}")
        answer.push("\t;")
        answer.push("END;")
        answer.push("\n")
        
        return answer.join("\n")
        
      end
