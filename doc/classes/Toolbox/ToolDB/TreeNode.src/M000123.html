      def fetch_hostgenes(organism_id,type)
        
        answer = []
        
        self.fetch_present_groups(type).each do |group|
          hostgene = group.fetch_hostgenes(organism_id).shift
          answer.push(hostgene) unless answer.include?(hostgene) or hostgene.nil?
        end
        
        return answer
        
      end
