      def overlaps_exon?
        
        return false unless self.has_hostgene?
        
        gene = self.fetch_hostgene.fetch_ensembl_gene
        exons = []
        gene.transcripts.each do |t|
          t.exons.each do |exon|
            return true if self.start >= exon.start and self.stop <= exon.stop
          end
          
        end
        
        return false
        
      end
