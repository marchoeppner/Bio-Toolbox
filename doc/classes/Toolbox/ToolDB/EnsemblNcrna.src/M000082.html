      def self.fetch_conserved(biotype)
        
        answer = Array.new
        ncrnas = ToolDB::EnsemblNcrna.find_all_by_biotype(biotype)
        ncrnas.each do |ncrna|
          answer.push(ncrna) if ncrna.has_group? 
        end
        
        return answer
        
      end
