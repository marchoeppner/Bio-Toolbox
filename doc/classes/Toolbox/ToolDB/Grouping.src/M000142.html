      def fetch_diff_scores(window_size=1,upstream=0,downstream=0)
        
        undefined = 0.0 # what to insert for columns without GERP scores
        
        valid_windows = [ 1 , 10, 100, 500 ]
        raise " Window size must be an integer and 1,10,100 or 500" unless valid_windows.include?(window_size)
        
        start = self.gal_start-upstream  # the position to be analyzed in alignment coordinates
        stop = self.gal_stop+downstream
        
        block = self.fetch_ensembl_genomic_align_block
        
        answer = [] # array of final scores
        bucket = {} # bin all conservation_scores falling in the region
        
        block.conservation_scores.select{|s| s.window_size == window_size and s.position < stop }.each do |score|
          
          diff_scores = score.get_diff_scores
          
          next unless score.position >= start and score.position <= stop or start >= score.position and start < (score.position+(diff_scores.nitems*window_size))
          
          bucket[score.position] = diff_scores
          
        end
        
        prev_pos = nil # keep track of the position in the reconstruction process and calculate gaps
        no_pos = (stop-start)/window_size # number of items (window size dependent) in this region to cover (used to exit loop)
        
        if bucket.to_a.empty? # if no GERP analysis was performed for this entire section...
          no_pos.times do 
            answer.push(undefined)
          end
          return answer
        end
        
        bucket.sort.each do |pos,scores|
          # subsequent scores in this position, added to the result - add 0.0 for gaps
          if prev_pos.nil? == false
            ((pos-prev_pos)/window_size).times do
              answer.push(undefined)
            end
            scores.each do |s|
              answer.push(s)
              return answer if answer.nitems == no_pos
            end
            prev_pos = (start+(answer.nitems*window_size)) 
          else
            # beginning of the score string - remove trailing scores and add to results
            if pos < start
              ((start-pos)/window_size).times do 
                scores.shift
              end
            elsif pos > start
              ((pos-start)/window_size).times do 
                answer.push(undefined)
              end
            end
            scores.each do |s|
              answer.push(s)
              return answer if answer.nitems == no_pos
            end
            prev_pos = (start+(answer.nitems*window_size))
          end
        end
        # if the number of scores is smaller then expected, fill up the remaining slots with 0
        if answer.nitems < ((stop-start)/window_size) 
          (((stop-start)-answer.nitems)/window_size).times do
            answer.push(undefined)
          end
        end 
        
        return answer
        
      end
