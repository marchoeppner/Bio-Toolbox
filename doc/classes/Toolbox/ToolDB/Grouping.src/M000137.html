      def fetch_unique_goterms
        
        answer = Array.new
        genes = self.fetch_hostgenes
        genes.each do |gene|
          gene.fetch_goterms.each do |go|
            answer.push(go.go_term) unless answer.include?(go.go_term)
          end
        end
        
        return answer
        
      end
