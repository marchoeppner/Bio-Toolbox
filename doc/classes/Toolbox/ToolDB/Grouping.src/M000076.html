      def fetch_hostgenes(organism_id=nil)
        
        answer = []
        
        if organism_id.nil?
          self.ensembl_ncrnas.each do |ncrna|
            answer.push(ncrna.fetch_hostgene) unless ncrna.fetch_hostgene.nil?
          end
        else
          self.ensembl_ncrnas.each do |ncrna|
            next if ncrna.organism_id != organism_id
            answer.push(ncrna.fetch_hostgene) unless ncrna.fetch_hostgene.nil?
          end
        end
        
        return answer
        
      end
