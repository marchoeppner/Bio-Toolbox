      def hierarchy(indent=1)
        
        if @hierarchy.nil?
          @hierarchy = Array.new
        end
        spacer = ""
        indent.times do 
          spacer = spacer + " "
        end
        @hierarchy.push("#{spacer}#{self.term}")
        
        if self.has_children?
          self.get_children.each do |child|
            
            child.hierarchy(indent+1)
          end
        end
        return @hierarchy
      end
