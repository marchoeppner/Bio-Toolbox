      def check_terms(list_of_terms,matched)
        
        # Check if the term has been identified as match before
        if matched.has_key?(self.id)
          return matched.fetch(self.id)
        end
        
        # if not, check this term for matches
        list_of_terms.each do |term|
          if self.name.include?("#{term}")
            return true 
          end
        end
        
        #ok, wasn't a match - remember that!
        matched[self.id] = false
        
        #else iterate over all parents...
        self.get_parents.each do |parent|
          parent.check_terms(list_of_terms,matched)
        end  
        
      end
