      def get_parents
        
        answer = []
        entries = ToolDB::GoToGo.find_all_by_go_term_child_id(self.id)
        entries.each do |entry|
          answer.push(ToolDB::GoTerm.find(entry.go_term_parent_id))
        end
        
        return answer
        
      end
