      def initialize(infile)
        @infile = infile
        @blocks = IO.readlines(infile)
        @header = @blocks.shift.split("|")
        @blocks.sort!.uniq!
        @dataset_id = @header.shift
        @type = @header.shift
        @organisms = @header.shift.split(",")
        @group_counter = 0
      end
