      def initialize(type,dataset_id,file,version=54)
        @type = type
        @dataset_id = dataset_id
        @outfile = "#{file}_all.xml"
        @version = version.to_i
        if @type == "snoRNA" or @type == "snRNA"
          @external_db = "RFAM"
        elsif @type == "miRNA"
          @external_db = "miRBase"
        end
      end
