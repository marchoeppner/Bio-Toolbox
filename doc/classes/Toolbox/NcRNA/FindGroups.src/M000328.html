      def print_alignment(group,aln,start,stop)
        puts "Printing alignment..."
        f = File.new("#{@base_name}/#{@type}_#{group}.fasta", "a")
        aln.each_pair do |k,seq|
          f.puts ">#{k}"
          f.puts "#{seq[start..stop]}"
        end
        f.close
      end
