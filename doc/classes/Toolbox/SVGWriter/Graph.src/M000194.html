                        def write
                                puts "<?xml version=\"1.0\" encoding=\"UTF-8\" standalone=\"no\"?>"
                                puts "<svg"
                                puts "\t<xmlns:dc=\"http://purl.org/dc/elements/1.1/\""
                        puts "\txmlns:cc=\"http://creativecommons.org/ns#\""
                        puts "\txmlns:rdf=\"http://www.w3.org/1999/02/22-rdf-syntax-ns#\""
                        puts "\txmlns:svg=\"http://www.w3.org/2000/svg\""
                        puts "\txmlns=\"http://www.w3.org/2000/svg\""
                                puts "\twidth=\"#{self.width}\""
                        puts "\theight=\"#{self.height}\">"
                        
                        self.elements.each do |element|
                        
                                element.write
                        
                        end
                        
                        end
