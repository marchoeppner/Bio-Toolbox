      def run
        
        IO.popen("RNAalifold #{options} < #{self.infile} 2>&1") do |f|
          while line = f.gets do
            @output.push(line)
          end
        end
        
        @consensus = @output[1]
        @structure = @output[2]
        
        system("rm #{self.infile}") if remove == true
        
      end
