      def initialize(alignment,options="",remove=true)
        
        @infile = "#{rand(100000)}_alifold.aln"
        
        f = File.new(@infile,"a")
        f.puts "#{alignment.raw}"
        f.close
        
        @alignment = alignment
        @options = options
        @remove = remove
        
        @output = []
        @forward = nil
        @reverse = nil
        
      end
