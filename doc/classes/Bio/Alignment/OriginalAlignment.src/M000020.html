      def run_baseml(tree)
        
        aln = Bio::Alignment::OriginalAlignment.new
        self.each_pair { |k,s| aln.store(k.gsub(/\s/,'')[0..6],s)}
        
        baseml = Bio::PAML::Baseml.new
        report = baseml.query(aln,tree)

        result = parse_baseml(baseml.output)
        return result
        
      end
