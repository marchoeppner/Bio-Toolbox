      def print_for_paml(tree)
        
        answer = []
        answer.push("\t#{self.number_of_sequences}\t#{self.alignment_length}")
        
        self.each_pair do |k,seq|
          answer.push("#{k.gsub(/\s/, '_')[0..9]}")
          while seq.length > 0
            answer.push("#{seq.slice!(0..29)}")
          end
        end
        
        return answer.join("\n")
      end
