      def gap_score
        
        return 0 if self.number_of_sequences == 1
        
        gaps = []
        
        self.each_site do |site|
          gaps.push((site.select{|s| "#{s}" == "-"}.nitems.to_f/site.nitems.to_f))
        end
        
        score = 0.0
        gaps.each {|g| score += g}
        return (score/self.alignment_length)*100
        
      end
