    def cigar_line
      
      answer = ""
      
      seq = "#{self.data.upcase.strip.gsub(/\n/, '')}"
      
      until seq.strip.length == 0
        if seq.match(/^-/)
          this_seq = seq.slice!(/^-+/)
          answer += "#{this_seq.length}D"
        else
          this_seq = seq.slice!(/^[A-Z\*]+/)
          answer += "#{this_seq.length}M"
        end
      end
      
      return answer
      
    end
