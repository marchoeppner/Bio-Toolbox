    def initialize(model,query_seqs,target_seqs,params=[])
          @model = model
          @query = "#{rand(1000)}_query.fasta"
          @target = "#{rand(1000)}_query.fasta"
          @params = params
          
          f = File.new("#{@query}","a")
          f.puts query_seqs.to_fasta
          f.close
          f = File.new("#{@target}", "a")
          f.puts target_seqs.to_fasta
          f.close
          
    end
