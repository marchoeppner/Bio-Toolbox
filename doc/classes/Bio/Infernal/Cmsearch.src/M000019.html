      def parse
                
                query = nil
                target = nil
                score = nil
                strand = nil
                
                answer = []
                this_match = {}
                
                self.output.each do |line|
            
                        if line.include?("Plus strand results")
                                this_match = { :strand => nil, :query => nil, :target => nil, :score => nil, :model => self.model}
                                strand = 1
                                this_match[:strand] = 1
                                
                        elsif line.include?("Minus strand results")
                                this_match = { :strand => nil, :query => nil, :target => nil, :score => nil, :model => self.model}
                                strand = 1
                                this_match[:strand] = -1
                                
                        elsif line.include?("Query")
                        
                                elements = line.split(",")
                                this_match[:query] = elements.shift.slice(/\d+\s-\s\d+/)
                                this_match[:target] = elements.shift.slice(/\d+\s-\s\d+/)
                        
                        elsif line.include?("Score")
                                
                                elements = line.split(",")
                                this_match[:score] = elements.shift.slice(/\d+\.\d+/)
                                puts "Match found: #{this_match[:score]}" if verbose
                                answer << this_match
                                this_match = { :strand => strand, :query => nil, :target => nil, :score => nil, :model => self.model }
                        else
                        
                        end
                        
                end
                
                return answer
                
        
      end
