      def get_alignment(type="ortholog")
        
        bin = []
        bin << self.to_fasta
        
        self.get_homology_seqs(type).each {|s| bin << s }
        
        aln = Bio::Alignment::OriginalAlignment.new(bin)
        
        factory = Bio::ClustalW.new
        a2 = aln.do_align(factory)
        
        return a2
        
      end
