      def fetch_noncoding
        
        answer = Array.new
    
        self.get_slice.genes.each do |gene|
          if gene.biotype == "protein_coding"
            gene.fetch_noncoding.each do |nc|
              answer.push(nc)   # a bunch of slice objects ...
            end
          end
        end
        
        return answer
        
      end
