      def structure_to_html(width=1024)
        
        snornas = self.snornas
        
        noncoding = self.fetch_noncoding
        coordinates = []
        
        sno_ranges = []
        snornas.each do |snorna|
          sno_ranges << Range.new(snorna.start,snorna.stop)
        end
               
        length = self.stop-self.start
        
        noncoding.each do |nc|
          coordinates << nc.start
          coordinates << nc.stop
        end
        
        coordinates.sort!
        background = { "exon" => "white", "intron" => "red", "snorna" => "blue" }
        
        answer = []
        answer << "<table border='1'>\n<tr>\n"
        type = "exon"
        
        previous = self.start
        until coordinates.empty?
          
          this_start = coordinates.shift
          this_stop = coordinates.shift

          this_length = ((this_start-previous).to_f/(length.to_f/100))#/100
          
          answer << "<td width='#{this_length}%' id='#{type}' bgcolor='#{background.fetch(type)}'></td>"
          type = "intron"
          
          this_length = ((this_stop-this_start).to_f/(length.to_f/100))
          sno_ranges.each do |snorna|
            type = "snorna" if snorna.first >= this_start and snorna.end <= this_stop
          end
          answer << "<td width='#{this_length}%' id='#{type}' bgcolor='#{background.fetch(type)}'></td>"
          type = "exon"
          
          previous = this_stop
        end
        
        this_length = ((self.stop-previous).to_f/(length.to_f/100))
        answer << "<td width='#{this_length}%' id='#{type}' bgcolor='#{background.fetch(type)}'></td>"
        
        answer << "</tr></table>"
        
        return answer.join("\n")
        
      end
