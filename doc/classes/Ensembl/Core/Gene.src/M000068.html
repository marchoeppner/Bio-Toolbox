      def fetch_hostgene
        return nil unless self.is_intronic?
        ncrna = self.slice
        genes = ncrna.genes(true).select{|g| g.stable_id != self.stable_id}
        genes.each do |gene|
          gene_slice = Slice.fetch_by_gene_stable_id(gene.stable_id)
          if ncrna.within?(gene_slice)
            return gene
          end
        end
      end
