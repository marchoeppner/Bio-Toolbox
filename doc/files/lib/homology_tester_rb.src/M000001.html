def do_align(group_id)

        puts "Aligning #{group_id} Thread: #{@count}"
        ff = Bio::FlatFile.open(Bio::FastaFormat,"#{group_id}_nt.fasta")
        
        aln = []
        
        ff.each_entry {|f| aln << f }
        
        a2 = Bio::Alignment.new(aln)
        
        factory = Bio::ClustalW.new
        
        result = a2.do_align(factory)
        f = File.new("#{group.id}_clustawl.fasta")
        result.each_pair do |name,seq|
        
                f.puts ">#{name}"
                f.puts "#{seq}"
        end
        f.close
end
