def run_probcons(this_id)

        puts "...Group #{this_id}, (Thread #{@count})"

        ff = Bio::FastaFormat.open("#{this_id}_aa.fasta")

        FileUtils.mkdir_p("#{this_id}")
        FileUtils.cd("#{this_id}") { |dir|
        
                ff.each_entry do |seq|
                        f = File.new("#{seq.definition}.fasta", "w+")
                        f.puts seq
                        f.close
                end
                
                system("probcons ../#{this_id}_aa.fasta > aligned_group.fasta")
        }

end
