def run_pecan(this_id)

        puts "...Group #{this_id}, (Thread #{@count})"

        ff = Bio::FastaFormat.open("#{this_id}_nt.fasta")

        FileUtils.mkdir_p("#{this_id}")
        FileUtils.cd("#{this_id}") { |dir|
        
                ff.each_entry do |seq|
                        f = File.new("#{seq.definition}.fasta", "w+")
                        f.puts seq
                        f.close
                end

                system("clustalw ../#{this_id}_nt.fasta -tree")
                tree_string = IO.readlines("../#{this_id}_nt.ph").collect{|l| l.strip.gsub(/\:(\d+)\.(\d+)/, '')}.join
                seqs =  tree_string.gsub(/[\(\);]/, '').split(",").collect{|s| s+".fasta"}

                system("java -cp /home/marc/scripts/pecan_v0.8/pecan_v0.8.jar bp.pecan.Pecan -E '#{tree_string}' -F #{seqs.join(" ")}")
        }
        

end
