def run_infernal(seq,models)
        
        answer = []
        
        models.each do |model|
                puts "Current process: #{seq.definition}, #{model} Thread: #{@count}"
                cm = Bio::Infernal::Cmsearch.new(seq,model)
                results = cm.run
                                
                results.each do |result|
                                                                        
                        strand = result[:strand]
                        target = result[:target].split(" - ")
                        start = target[0]
                        stop = target[1]
                        query = result[:query]
                        score = result[:score]
                                        
                        answer << "INSERT INTO infernal_hit(member_id,strand,member_start,member_stop,score,rfam_id) VALUES((SELECT id FROM member WHERE stable_id = '#{seq.definition}'),#{strand},#{start},#{stop},#{score},(SELECT id FROM rfam WHERE accession = '#{model}'));"
                end
                                
        end
        
        return answer
        
end
